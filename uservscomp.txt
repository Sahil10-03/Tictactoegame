void input()
{

    if(player==val)
        cout<<"Player 1 turn:"<<endl;
    else
    cout<<"Player 2 turn:"<<endl;
    cout<<"Select the number of the box which you want to mark"<<endl;
    int n;
    cin>>n;

    if(n>9||n<1)
    {
        cout<<"Out of Range box selected. Please chjoose again"<<endl;
        input();
        return ;
    }
    if(arr[n-1]=='X'||arr[n-1]=='O')
    {
        cout<<"This position is already marked, choose again"<<endl;
        input();
        return;
    }
    arr[n-1]=char(player);

}


void compinput()
{

    int comp_choice=rand()%9;

    if(comp_choice<1||comp_choice>9)
    {
        compinput();
        return ;
    }
    if(arr[comp_choice]=='X'||arr[comp_choice]=='O')
    {
        compinput();
        return ;

    }
    if(player=='X')
        arr[comp_choice]='O';
        else
            arr[comp_choice]='X';



}

void toggleplayer()
{
    if(player=='X')
